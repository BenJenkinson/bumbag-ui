```jsx
import { ProgressBar } from 'fannypack';
```


### Animation

```function-live
function Example() {
  const [value, setValue] = React.useState(30);

  return (
    <Stack>
      <Set>
        <Button
          onClick={() => setValue(value => value - 10)}
        >
          -10
        </Button>
        <Button
          onClick={() => setValue(value => value + 10)}
        >
          +10
        </Button>
      </Set>
      <ProgressBar value={value} />
    </Stack>
  );
}
```
