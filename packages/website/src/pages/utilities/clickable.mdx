## Usage

```jsx-live
<Set>
  <Clickable
    use={Button}
    onClick={() => alert('Clicked')}
  >
    Clickable
  </Clickable>
  <Clickable
    use={Button}
    disabled
    onClick={() => alert('Clicked')}
  >
    Clickable
  </Clickable>
  <Clickable
    use={Button}
    disabled
    focusable
    onClick={() => alert('Clicked')}
  >
    Clickable
  </Clickable>
</Set>
```
