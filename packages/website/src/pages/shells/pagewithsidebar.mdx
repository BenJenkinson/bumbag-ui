# PageWithSidebar

## Import

```jsx
import { Page } from 'fannypack';
```

## Usage

```jsx
<PageWithSidebar sidebar={<Box>This is a sidebar</Box>}>
  Hello world
</PageWithSidebar>
```

### Collapse breakpoint

```jsx
<PageWithSidebar
  collapseBreakpoint="max-desktop"
  sidebar={<Box>This is a sidebar</Box>}
>
  Hello world
</PageWithSidebar>
```

### Toggling the sidebar

```jsx-live
<PageWithSidebar.Disclosure>
  Toggle sidebar
</PageWithSidebar.Disclosure>
```

### Minimizing the sidebar

```jsx-live
<PageWithSidebar.Minimize>
  Minimize sidebar
</PageWithSidebar.Minimize>
```

### Sidebar width

#### Expanded

```jsx
<PageWithSidebar
  sidebar={<Box>This is a sidebar</Box>}
  sidebarWidth="250px"
>
  This is the content
</PageWithSidebar>
```

#### Collapsed

```jsx
<PageWithSidebar
  sidebar={<Box>This is a sidebar</Box>}
  collapsedSidebarWidth="300px"
>
  This is the content
</PageWithSidebar>
```

#### Minimized

```jsx
<PageWithSidebar
  sidebar={<Box>This is a sidebar</Box>}
  minimizedSidebarWidth="300px"
>
  This is the content
</PageWithSidebar>
```
